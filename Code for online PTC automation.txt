;=================================== BeizerMouseCurve =========================================
;==============================================================================================

BeizerMouseCurve(px, py)
{
   t = 0
   Random, x3, 0, 1300
   Random, y3, 0, 700
   Random, x4, 0, 1300
   Random, y4, 0, 700
   while(t <= 1)
   {
      MouseGetPos, x0, y0
      ax := Abs(x0 - px)
      ay := Abs(y0 - py)
      if (ax < 10) and (ay < 10)
      {
         r = 5
      }
      else
      {
         r = 10
      }
      Random, x, 1, %r%
      x1 = % x0 + x
      Random, x, 1, %r%
      y1 = % y0 + x
      Random, x, 1, %r%
      x2 = % x1 + x
      Random, x, 1, %r%
      y2 = % y1 + x
      pfinalx := (1 - t)*(1 - t)*(1 - t)*(1 - t)*(1 - t)*x0 + 5*(1 - t)*(1 - t)*(1 - t)*(1 - t)*t*x1 + 10*(1 - t)*(1 - t)*(1 - t)*t*t*x2 + 10*t*t*t*(1 - t)*(1 - t)*x3 + 5*t*t*t*t*(1 - t)*x4 + t*t*t*t*t*px
      pfinaly := (1 - t)*(1 - t)*(1 - t)*(1 - t)*(1 - t)*y0 + 5*(1 - t)*(1 - t)*(1 - t)*(1 - t)*t*y1 + 10*(1 - t)*(1 - t)*(1 - t)*t*t*y2 + 10*t*t*t*(1 - t)*(1 - t)*y3 + 5*t*t*t*t*(1 - t)*y4 + t*t*t*t*t*py
      MouseMove, %pfinalx%, %pfinaly%
      t += 0.01
   }
   t = 0
   while(t <= 1)
   {
      MouseGetPos, x0, y0
      Random, x, 0, 1
      x1 = % x0 + x
      Random, x, 0, 1
      y1 = % y0 + x
      Random, x, 0, 1
      x2 = % x1 + x
      Random, x, 0, 1
      y2 = % y1 + x
      pfinalx := (1 - t)*(1 - t)*(1 - t)*x0 + 3*(1 - t)*(1 - t)*t*x1 + 3*(1 - t)*t*t*x2 + t*t*t*px
      pfinaly := (1 - t)*(1 - t)*(1 - t)*y0 + 3*(1 - t)*(1 - t)*t*y1 + 3*(1 - t)*t*t*y2 + t*t*t*py
      Random, s, 1, 20
      Sleep, %s%
      MouseMove, %pfinalx%, %pfinaly%
      t += 0.1
   }
   Random, s, 300, 600
   Sleep, %s%
   SendInput {Click}
   return
}
